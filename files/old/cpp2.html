const int N = 1003;
int t[N][N];
	
int v, e, x, y;
cin >> v >> e;
for(int i = 0; i < e; i++){
	cin >> x >> y;
	t[x][y] = 1;
}

Mnożenie macierzy O(N*M*R)

const int N = 1003, M = 1004, R = 1005;
int a[N][M], b[M][R], c[N][R];

for(int i = 0; i < N; i++)
	for(int j = 0; j < R; j++) {
		c[i][j] = 0;
		for(int k = 0; k < M; k++)
			c[i][j] += a[i][k]*b[k][j];
	}

Algorytm na liczbę ścieżek długości k z v1 do v2. O(k * |V|3)
